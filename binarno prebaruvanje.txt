import java.util.Scanner;

class BinarySearch {
    int binarySearch(int niza[], int left, int right, int key) {
        if (left > right) {
            return -1;
        }
        int mid = (left + right) / 2;

        if (niza[mid] == key) {
            return mid;
        } else if (niza[mid] > key) {
            return binarySearch(niza, left, mid - 1, key);
        } else {
            return binarySearch(niza, mid + 1, right, key);
        }
    }
}

public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        BinarySearch bs = new BinarySearch();
        int n = scanner.nextInt();
        int key = scanner.nextInt();
        int niza[] = new int[n];
        for (int i = 0; i < n; i++) {
            niza[i] = scanner.nextInt();
        }

        int rezultat = bs.binarySearch(niza, 0, n - 1, key);
        if (rezultat == -1) {
            System.out.println("Ne postoi");
        } else {
            System.out.println(rezultat);
        }
    }
}
