import java.util.*;
import java.io.*;

public class Main {
    public static void main(String[] args) throws IOException {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int[] a = new int[n];
        for (int i = 0; i < n; i++) {
            a[i] = sc.nextInt();
        }

        int[] zalutani = new int[n];
        int[] pomestuvanje = new int[n];
        int brojac = 0;
        int max = Integer.MIN_VALUE;

        for (int i = 0; i < n; i++) {
            if (a[i] < max) {
                int tocenIndex = bynarySearch(a, 0, i - 1, a[i]);
                zalutani[brojac] = a[i];
                pomestuvanje[brojac] = i - tocenIndex;
                brojac++;
            } else {
                max = a[i];
            }
        }

        System.out.println(brojac);
        for (int i = 0; i < brojac; i++) {
            System.out.println(zalutani[i] + " " + pomestuvanje[i]);
        }
    }


    private static int bynarySearch(int[] a, int pocetok, int kraj, int k) {
        if (pocetok > kraj) {
            return pocetok;
        }

        int sredina = (pocetok + kraj) / 2;

        if (a[sredina] > k) {
            return bynarySearch(a, pocetok, sredina - 1, k);
        } else {
            return bynarySearch(a, sredina + 1, kraj, k);
        }
    }
}
